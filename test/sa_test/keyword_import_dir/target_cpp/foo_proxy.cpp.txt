#include "foo_proxy.h"

namespace test {
FooProxy::FooProxy(const sptr<IRemoteObject>& remote) : IRemoteProxy<IFoo>(remote)
{
    std::lock_guard<std::mutex> lock(mutex_);
    if (remote_ && remote_ != remote) {
        RemoveDeathRecipient();
    }
    if (remote) {
        if (!remote->IsProxyObject()) {
            return;
        }
        deathRecipient_ = new (std::nothrow) FooRecipient(*this);
        if (!deathRecipient_) {
            return;
        }
        if (!remote->AddDeathRecipient(deathRecipient_)) {
            return;
        }
        remote_ = remote;
    }
}

FooProxy::~FooProxy()
{
    RemoveDeathRecipient();
}

void FooProxy::RemoveDeathRecipient()
{
    std::lock_guard<std::mutex> lock(mutex_);
    if (!remote_ || !deathRecipient_) {
        return;
    }
    remote_->RemoveDeathRecipient(deathRecipient_);
    remote_ = nullptr;
    deathRecipient_ = nullptr;
    remoteDiedCallback_ = nullptr;
}

void FooProxy::OnRemoteDied(const wptr<IRemoteObject>& remote)
{
    if (remoteDiedCallback_) {
        remoteDiedCallback_(remote);
    }
    RemoveDeathRecipient();
}

ErrCode FooProxy::seq_test_func(
    const myseq& inParam,
    myseq& outParam,
    myseq* inoutParam)
{
    MessageParcel data;
    MessageParcel reply;
    MessageOption option(MessageOption::TF_SYNC);

    if (!data.WriteInterfaceToken(GetDescriptor())) {
        return ERR_INVALID_VALUE;
    }

    if (!data.WriteParcelable(&inParam)) {
        return ERR_INVALID_DATA;
    }
    if (!data.WriteParcelable(&inoutParam)) {
        return ERR_INVALID_DATA;
    }

    sptr<IRemoteObject> remote = Remote();
    if (!remote) {
        return ERR_INVALID_DATA;
    }
    int32_t result = remote->SendRequest(
        static_cast<uint32_t>(IFooIpcCode::COMMAND_SEQ_TEST_FUNC), data, reply, option);
    if (FAILED(result)) {
        return result;
    }

    ErrCode errCode = reply.ReadInt32();
    if (FAILED(errCode)) {
        return errCode;
    }

    std::unique_ptr<myseq> outParamInfo(reply.ReadParcelable<myseq>());
    if (outParamInfo != nullptr) {
        outParam = *outParamInfo;
    }

    std::unique_ptr<myseq> inoutParamInfo(reply.ReadParcelable<myseq>());
    if (inoutParamInfo != nullptr) {
        inoutParam = *inoutParamInfo;
    }

    return ERR_OK;
}

ErrCode FooProxy::interface_test_func(
    const sptr<myinterface2>& inParam,
    sptr<myinterface2>& outParam,
    const sptr<myinterface2>& inoutParam)
{
    MessageParcel data;
    MessageParcel reply;
    MessageOption option(MessageOption::TF_SYNC);

    if (!data.WriteInterfaceToken(GetDescriptor())) {
        return ERR_INVALID_VALUE;
    }

    if (inParam == nullptr) {
        return ERR_INVALID_DATA;
    }
    if (!data.WriteRemoteObject(inParam->AsObject())) {
        return ERR_INVALID_DATA;
    }
    if (inoutParam == nullptr) {
        return ERR_INVALID_DATA;
    }
    if (!data.WriteRemoteObject(inoutParam->AsObject())) {
        return ERR_INVALID_DATA;
    }

    sptr<IRemoteObject> remote = Remote();
    if (!remote) {
        return ERR_INVALID_DATA;
    }
    int32_t result = remote->SendRequest(
        static_cast<uint32_t>(IFooIpcCode::COMMAND_INTERFACE_TEST_FUNC), data, reply, option);
    if (FAILED(result)) {
        return result;
    }

    ErrCode errCode = reply.ReadInt32();
    if (FAILED(errCode)) {
        return errCode;
    }

    outParam = iface_cast<myinterface2>(reply.ReadRemoteObject());
    inoutParam = iface_cast<myinterface2>(reply.ReadRemoteObject());
    return ERR_OK;
}
