#ifndef OHOS_HDI_PIN_AUTH_V1_0_EXECUTORPROXY_H
#define OHOS_HDI_PIN_AUTH_V1_0_EXECUTORPROXY_H

#include "v1_0/iexecutor.h"
#include <unistd.h>
#include <iproxy_broker.h>

namespace OHOS {
namespace HDI {
namespace PinAuth {
namespace V1_0 {

class ExecutorProxy : public IProxyBroker<OHOS::HDI::PinAuth::V1_0::IExecutor> {
public:
    class IServiceManagerDeathRecipient : public IRemoteObject::DeathRecipient {
    public:
        IServiceManagerDeathRecipient(wptr<OHOS::HDI::PinAuth::V1_0::ExecutorProxy> proxy) : proxy_(proxy) {} 
        ~IServiceManagerDeathRecipient() override = default;
        void OnRemoteDied(const wptr<IRemoteObject> &remote) override
        {
            int32_t result = HDF_FAILURE;
            const int sleepInterval = 500000;
            const int waitTimes = 10;
            int currentTime = waitTimes;
            do {
                usleep(sleepInterval);
                auto proxy = proxy_.promote();
                if (proxy != nullptr) {
                    result = OHOS::HDI::PinAuth::V1_0::ExecutorProxy::Reconnect(proxy);
                }
                --currentTime;
            } while (result != HDF_SUCCESS && currentTime >0);
        }
    private:
        wptr<OHOS::HDI::PinAuth::V1_0::ExecutorProxy> proxy_;
    };

    explicit ExecutorProxy(const sptr<IRemoteObject>& remote) : IProxyBroker<OHOS::HDI::PinAuth::V1_0::IExecutor>(remote) {
        reconnectRemote_ = nullptr;
        isReconnected_ = false;
    }
    virtual ~ExecutorProxy() = default;

    inline bool IsProxy() override
    {
        return true;
    }

    int32_t GetExecutorInfo(OHOS::HDI::PinAuth::V1_0::ExecutorInfo& executorInfo) override;

    int32_t GetTemplateInfo(uint64_t templateId, OHOS::HDI::PinAuth::V1_0::TemplateInfo& templateInfo) override;

    int32_t OnRegisterFinish(const std::vector<uint64_t>& templateIdList,
         const std::vector<uint8_t>& frameworkPublicKey, const std::vector<uint8_t>& extraInfo) override;

    int32_t OnSetData(uint64_t scheduleId, uint64_t authSubType, const std::vector<uint8_t>& data) override;

    int32_t Enroll(uint64_t scheduleId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj) override;

    int32_t Authenticate(uint64_t scheduleId, uint64_t templateId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj) override;

    int32_t Delete(uint64_t templateId) override;

    int32_t Cancel(uint64_t scheduleId) override;

    int32_t SendCommand(int32_t commandId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj) override;

    int32_t GetVersion(uint32_t& majorVer, uint32_t& minorVer) override;

    static int32_t GetExecutorInfo_(OHOS::HDI::PinAuth::V1_0::ExecutorInfo& executorInfo,
         const sptr<IRemoteObject> remote);

    static int32_t GetTemplateInfo_(uint64_t templateId, OHOS::HDI::PinAuth::V1_0::TemplateInfo& templateInfo,
         const sptr<IRemoteObject> remote);

    static int32_t OnRegisterFinish_(const std::vector<uint64_t>& templateIdList,
         const std::vector<uint8_t>& frameworkPublicKey, const std::vector<uint8_t>& extraInfo, const sptr<IRemoteObject> remote);

    static int32_t OnSetData_(uint64_t scheduleId, uint64_t authSubType, const std::vector<uint8_t>& data,
         const sptr<IRemoteObject> remote);

    static int32_t Enroll_(uint64_t scheduleId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj, const sptr<IRemoteObject> remote);

    static int32_t Authenticate_(uint64_t scheduleId, uint64_t templateId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj, const sptr<IRemoteObject> remote);

    static int32_t Delete_(uint64_t templateId, const sptr<IRemoteObject> remote);

    static int32_t Cancel_(uint64_t scheduleId, const sptr<IRemoteObject> remote);

    static int32_t SendCommand_(int32_t commandId, const std::vector<uint8_t>& extraInfo,
         const sptr<OHOS::HDI::PinAuth::V1_0::IExecutorCallback>& callbackObj, const sptr<IRemoteObject> remote);

    static int32_t GetVersion_(uint32_t& majorVer, uint32_t& minorVer, const sptr<IRemoteObject> remote);

    static int32_t Reconnect(sptr<OHOS::HDI::PinAuth::V1_0::ExecutorProxy> proxy);

    sptr<IRemoteObject> GetCurrentRemote() {
        return isReconnected_ ? reconnectRemote_ : Remote();
    }

    bool isReconnected_;
    std::string serviceName_;
    sptr<IRemoteObject> servMgr_;
    sptr<IRemoteObject> reconnectRemote_;
private:
    static inline BrokerDelegator<OHOS::HDI::PinAuth::V1_0::ExecutorProxy> delegator_;
};

} // V1_0
} // PinAuth
} // HDI
} // OHOS

#endif // OHOS_HDI_PIN_AUTH_V1_0_EXECUTORPROXY_H
