sptr<OHOS::HDI::Vibrator::V1_0::IVibratorInterface> OHOS::HDI::Vibrator::V1_0::IVibratorInterface::Get(const std::string& serviceName, bool isStub)
{
    if (!isStub) {
        return nullptr;
    }
    std::string desc = Str16ToStr8(OHOS::HDI::Vibrator::V1_0::IVibratorInterface::GetDescriptor());
    void *impl = LoadHdiImpl(desc.c_str(), serviceName == "vibrator_interface_service" ? "service" : serviceName.c_str());
    if (impl == nullptr) {
        HDF_LOGE("failed to load hdi impl %{public}s", desc.c_str());
        return nullptr;
    }
    return reinterpret_cast<OHOS::HDI::Vibrator::V1_0::IVibratorInterface *>(impl);
}

int32_t OHOS::HDI::Vibrator::V1_0::VibratorInterfaceStub::OnRemoteRequest(uint32_t code, MessageParcel& data, MessageParcel& reply, MessageOption& option)
{
    switch (code) {
        case CMD_VIBRATOR_INTERFACE_GET_VERSION:
            return VibratorInterfaceStubGetVersion(data, reply, option);
        case CMD_VIBRATOR_INTERFACE_START_ONCE:
            return VibratorInterfaceStubStartOnce(data, reply, option);
        case CMD_VIBRATOR_INTERFACE_START:
            return VibratorInterfaceStubStart(data, reply, option);
        case CMD_VIBRATOR_INTERFACE_STOP:
            return VibratorInterfaceStubStop(data, reply, option);
        default: {
            HDF_LOGE("%{public}s: cmd %{public}d is not supported", __func__, code);
            return IPCObjectStub::OnRemoteRequest(code, data, reply, option);
        }
    }
}
