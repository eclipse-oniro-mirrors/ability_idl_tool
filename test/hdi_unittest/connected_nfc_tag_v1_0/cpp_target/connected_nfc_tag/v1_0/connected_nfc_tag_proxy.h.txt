#ifndef OHOS_HDI_CONNECTED_NFC_TAG_V1_0_CONNECTEDNFCTAGPROXY_H
#define OHOS_HDI_CONNECTED_NFC_TAG_V1_0_CONNECTEDNFCTAGPROXY_H

#include "v1_0/iconnected_nfc_tag.h"
#include <unistd.h>
#include <iproxy_broker.h>

namespace OHOS {
namespace HDI {
namespace ConnectedNfcTag {
namespace V1_0 {

class ConnectedNfcTagProxy : public IProxyBroker<OHOS::HDI::ConnectedNfcTag::V1_0::IConnectedNfcTag> {
public:
    class IServiceManagerDeathRecipient : public IRemoteObject::DeathRecipient {
    public:
        IServiceManagerDeathRecipient(wptr<OHOS::HDI::ConnectedNfcTag::V1_0::ConnectedNfcTagProxy> proxy) : proxy_(proxy) {} 
        ~IServiceManagerDeathRecipient() override = default;
        void OnRemoteDied(const wptr<IRemoteObject> &remote) override
        {
            int32_t result = HDF_FAILURE;
            const int sleepInterval = 500000;
            const int waitTimes = 10;
            int currentTime = waitTimes;
            do {
                usleep(sleepInterval);
                auto proxy = proxy_.promote();
                if (proxy != nullptr) {
                    result = OHOS::HDI::ConnectedNfcTag::V1_0::ConnectedNfcTagProxy::Reconnect(proxy);
                }
                --currentTime;
            } while (result != HDF_SUCCESS && currentTime >0);
        }
    private:
        wptr<OHOS::HDI::ConnectedNfcTag::V1_0::ConnectedNfcTagProxy> proxy_;
    };

    explicit ConnectedNfcTagProxy(const sptr<IRemoteObject>& remote) : IProxyBroker<OHOS::HDI::ConnectedNfcTag::V1_0::IConnectedNfcTag>(remote) {
        reconnectRemote_ = nullptr;
        isReconnected_ = false;
    }
    virtual ~ConnectedNfcTagProxy() = default;

    inline bool IsProxy() override
    {
        return true;
    }

    int32_t Init() override;

    int32_t Uninit() override;

    int32_t ReadNdefTag(std::string& ndefData) override;

    int32_t WriteNdefTag(const std::string& ndefData) override;

    int32_t GetVersion(uint32_t& majorVer, uint32_t& minorVer) override;

    static int32_t Init_(const sptr<IRemoteObject> remote);

    static int32_t Uninit_(const sptr<IRemoteObject> remote);

    static int32_t ReadNdefTag_(std::string& ndefData, const sptr<IRemoteObject> remote);

    static int32_t WriteNdefTag_(const std::string& ndefData, const sptr<IRemoteObject> remote);

    static int32_t GetVersion_(uint32_t& majorVer, uint32_t& minorVer, const sptr<IRemoteObject> remote);

    static int32_t Reconnect(sptr<OHOS::HDI::ConnectedNfcTag::V1_0::ConnectedNfcTagProxy> proxy);

    sptr<IRemoteObject> GetCurrentRemote() {
        return isReconnected_ ? reconnectRemote_ : Remote();
    }

    bool isReconnected_;
    std::string serviceName_;
    sptr<IRemoteObject> servMgr_;
    sptr<IRemoteObject> reconnectRemote_;
private:
    static inline BrokerDelegator<OHOS::HDI::ConnectedNfcTag::V1_0::ConnectedNfcTagProxy> delegator_;
};

} // V1_0
} // ConnectedNfcTag
} // HDI
} // OHOS

#endif // OHOS_HDI_CONNECTED_NFC_TAG_V1_0_CONNECTEDNFCTAGPROXY_H
