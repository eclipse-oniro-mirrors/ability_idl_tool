static bool WritePodArray(struct HdfSBuf *parcel, const void *data, uint32_t elementSize, uint32_t count);


static bool WritePodArray(struct HdfSBuf *parcel, const void *data, uint32_t elementSize, uint32_t count)
{
    if (!HdfSbufWriteUint32(parcel, count)) {
        HDF_LOGE("%{public}s: failed to write array size", __func__);
        return false;
    }

    if (data == NULL && count == 0) {
        return true;
    }

    if (!HdfSbufWriteUnpadBuffer(parcel, (const uint8_t *)data, elementSize * count)) {
        HDF_LOGE("%{public}s: failed to write array", __func__);
        return false;
    }

    return true;
}
static int32_t SerStubCaptureFrame(struct IAudioCapture *serviceImpl, struct HdfSBuf *audioCaptureData, struct HdfSBuf *audioCaptureReply)
{
    int32_t audioCaptureRet = HDF_FAILURE;
    bool audioCaptureMemSet = false;
    int8_t* frame = NULL;
    uint32_t frameLen = 0;
    uint64_t replyBytes = 0;

    if (!HdfSbufReadUint8(audioCaptureData, (uint8_t *)&audioCaptureMemSet)) {
        HDF_LOGE("%{public}s: read flag of memory setting failed!", __func__);
        audioCaptureRet = HDF_ERR_INVALID_PARAM;
        goto FINISHED;
    }

    if (audioCaptureMemSet) {
        if (!HdfSbufReadUint32(audioCaptureData, &frameLen)) {
            HDF_LOGE("%{public}s: read frame size failed!", __func__);
            audioCaptureRet = HDF_ERR_INVALID_PARAM;
            goto FINISHED;
        }

        HDI_CHECK_VALUE_RET_GOTO(frameLen, >, HDI_BUFF_MAX_SIZE / sizeof(int8_t), audioCaptureRet, HDF_ERR_INVALID_PARAM, FINISHED);
        if (frameLen > 0) {
            frame = (int8_t*)OsalMemCalloc(sizeof(int8_t) * (frameLen));
            if (frame == NULL) {
                HDF_LOGE("%{public}s: malloc frame failed", __func__);
                audioCaptureRet = HDF_ERR_MALLOC_FAIL;
                goto FINISHED;
            }
        }
    } else {
        frame = (int8_t*)OsalMemCalloc(HDI_BUFF_MAX_SIZE);
        if (frame == NULL) {
            HDF_LOGE("%{public}s: malloc frame failed", __func__);
            audioCaptureRet = HDF_ERR_MALLOC_FAIL;
            goto FINISHED;
        }
        frameLen = (HDI_BUFF_MAX_SIZE / sizeof(int8_t));
    }


    if (serviceImpl == NULL) {
        HDF_LOGE("%{public}s: invalid serviceImpl object", __func__);
        audioCaptureRet = HDF_ERR_INVALID_OBJECT;
        goto FINISHED;
    }

    if (serviceImpl->CaptureFrame == NULL) {
        HDF_LOGE("%{public}s: invalid interface function CaptureFrame ", __func__);
        audioCaptureRet = HDF_ERR_NOT_SUPPORT;
        goto FINISHED;
    }

    audioCaptureRet = serviceImpl->CaptureFrame(serviceImpl, frame, &frameLen, &replyBytes);
    if (audioCaptureRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, audioCaptureRet);
        goto FINISHED;
    }

    if (!WritePodArray(audioCaptureReply, frame, sizeof(int8_t), frameLen)) {
        HDF_LOGE("%{public}s: failed to write frame", __func__);
        audioCaptureRet = HDF_ERR_INVALID_PARAM;
        goto FINISHED;
    }

    if (!HdfSbufWriteUint64(audioCaptureReply, replyBytes)) {
        HDF_LOGE("%{public}s: write replyBytes failed!", __func__);
        audioCaptureRet = HDF_ERR_INVALID_PARAM;
        goto FINISHED;
    }

FINISHED:
    if (frame != NULL) {
        OsalMemFree(frame);
    }
    return audioCaptureRet;
}

static struct HdfRemoteService **AudioCaptureStubNewInstance(void *impl)
{
    if (impl == NULL) {
        HDF_LOGE("%{public}s: impl is null", __func__);
        return NULL;
    }

    struct IAudioCapture *serviceImpl = (struct IAudioCapture *)impl;
    struct AudioCaptureStub *stub = OsalMemCalloc(sizeof(struct AudioCaptureStub));
    if (stub == NULL) {
        HDF_LOGE("%{public}s: failed to malloc stub object", __func__);
        return NULL;
    }
    stub->remote = HdfRemoteServiceObtain((struct HdfObject *)stub, &stub->dispatcher);
    if (stub->remote == NULL) {
        OsalMemFree(stub);
        return NULL;
    }
    (void)HdfRemoteServiceSetInterfaceDesc(stub->remote, IAUDIOCAPTURE_INTERFACE_DESC);
    stub->dispatcher.Dispatch = AudioCaptureOnRemoteRequest;
    stub->interface = serviceImpl;
    stub->interface->AsObject = AudioCaptureStubAsObject;
    return &stub->remote;
}

static void AudioCaptureStubRelease(struct HdfRemoteService **remote)
{
    if (remote == NULL) {
        return;
    }
    struct AudioCaptureStub *stub = CONTAINER_OF(remote, struct AudioCaptureStub, remote);
    HdfRemoteServiceRecycle(stub->remote);
    OsalMemFree(stub);
}
