sptr<OHOS::HDI::Ril::V1_0::IRil> OHOS::HDI::Ril::V1_0::IRil::Get(bool isStub)
{
    return OHOS::HDI::Ril::V1_0::IRil::Get("ril_service", isStub);
}

sptr<OHOS::HDI::Ril::V1_0::IRil> OHOS::HDI::Ril::V1_0::IRil::Get(const std::string& serviceName, bool isStub)
{
    if (!isStub) {
        return nullptr;
    }
    std::string desc = Str16ToStr8(OHOS::HDI::Ril::V1_0::IRil::GetDescriptor());
    void *impl = LoadHdiImpl(desc.c_str(), serviceName == "ril_service" ? "service" : serviceName.c_str());
    if (impl == nullptr) {
        HDF_LOGE("failed to load hdi impl %{public}s", desc.c_str());
        return nullptr;
    }
    return reinterpret_cast<OHOS::HDI::Ril::V1_0::IRil *>(impl);
}


OHOS::HDI::Ril::V1_0::RilStub::RilStub(const sptr<OHOS::HDI::Ril::V1_0::IRil> &impl)
    : IPCObjectStub(OHOS::HDI::Ril::V1_0::IRil::GetDescriptor()), impl_(impl)
{
}

OHOS::HDI::Ril::V1_0::RilStub::~RilStub()
{
    HDF_LOGI("%{public}s enter", __func__);
    ObjectCollector::GetInstance().RemoveObject(impl_);
}
