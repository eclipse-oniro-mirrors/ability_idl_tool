#ifndef OHOS_HDI_LOCATION_LPFENCE_CELLBATCHING_V1_0_CELLBATCHINGINTERFACEPROXY_H
#define OHOS_HDI_LOCATION_LPFENCE_CELLBATCHING_V1_0_CELLBATCHINGINTERFACEPROXY_H

#include "v1_0/icellbatching_interface.h"
#include <unistd.h>
#include <iproxy_broker.h>

namespace OHOS {
namespace HDI {
namespace Location {
namespace Lpfence {
namespace Cellbatching {
namespace V1_0 {

class CellbatchingInterfaceProxy : public IProxyBroker<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingInterface> {
public:
    class IServiceManagerDeathRecipient : public IRemoteObject::DeathRecipient {
    public:
        IServiceManagerDeathRecipient(wptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingInterfaceProxy> proxy) : proxy_(proxy) {} 
        ~IServiceManagerDeathRecipient() override = default;
        void OnRemoteDied(const wptr<IRemoteObject> &remote) override
        {
            int32_t result = HDF_FAILURE;
            const int sleepInterval = 500000;
            const int waitTimes = 10;
            int currentTime = waitTimes;
            do {
                usleep(sleepInterval);
                auto proxy = proxy_.promote();
                if (proxy != nullptr) {
                    result = OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingInterfaceProxy::Reconnect(proxy);
                }
                --currentTime;
            } while (result != HDF_SUCCESS && currentTime >0);
        }
    private:
        wptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingInterfaceProxy> proxy_;
    };

    explicit CellbatchingInterfaceProxy(const sptr<IRemoteObject>& remote) : IProxyBroker<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingInterface>(remote) {
        reconnectRemote_ = nullptr;
        isReconnected_ = false;
    }
    virtual ~CellbatchingInterfaceProxy() = default;

    inline bool IsProxy() override
    {
        return true;
    }

    int32_t RegisterCellBatchingCallback(const sptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingCallback>& callbackObj) override;

    int32_t UnregisterCellBatchingCallback(const sptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingCallback>& callbackObj) override;

    int32_t CellbatchingSwitch(const OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingRequest& req) override;

    int32_t FlushCellbatching() override;

    int32_t GetVersion(uint32_t& majorVer, uint32_t& minorVer) override;

    static int32_t RegisterCellBatchingCallback_(const sptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingCallback>& callbackObj, const sptr<IRemoteObject> remote);

    static int32_t UnregisterCellBatchingCallback_(const sptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::ICellbatchingCallback>& callbackObj, const sptr<IRemoteObject> remote);

    static int32_t CellbatchingSwitch_(const OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingRequest& req,
         const sptr<IRemoteObject> remote);

    static int32_t FlushCellbatching_(const sptr<IRemoteObject> remote);

    static int32_t GetVersion_(uint32_t& majorVer, uint32_t& minorVer, const sptr<IRemoteObject> remote);

    static int32_t Reconnect(sptr<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingInterfaceProxy> proxy);

    sptr<IRemoteObject> GetCurrentRemote() {
        return isReconnected_ ? reconnectRemote_ : Remote();
    }

    bool isReconnected_;
    std::string serviceName_;
    sptr<IRemoteObject> servMgr_;
    sptr<IRemoteObject> reconnectRemote_;
private:
    static inline BrokerDelegator<OHOS::HDI::Location::Lpfence::Cellbatching::V1_0::CellbatchingInterfaceProxy> delegator_;
};

} // V1_0
} // Cellbatching
} // Lpfence
} // Location
} // HDI
} // OHOS

#endif // OHOS_HDI_LOCATION_LPFENCE_CELLBATCHING_V1_0_CELLBATCHINGINTERFACEPROXY_H
