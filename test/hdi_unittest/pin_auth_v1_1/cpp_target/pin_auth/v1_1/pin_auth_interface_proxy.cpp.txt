template<typename ChildType, typename ParentType>
static sptr<ChildType> CastFromTemplate(const sptr<ParentType> &parent)
{
    if (parent == nullptr) {
        HDF_LOGE("%{public}s:parent is nullptr!", __func__);
        return nullptr;
    }

    if (!parent->IsProxy()) {
        HDF_LOGE("%{public}s:not proxy, not support castfrom!", __func__);
        return nullptr;
    }

    sptr<IRemoteObject> remote = OHOS::HDI::hdi_objcast<ParentType>(parent);
    if (remote == nullptr) {
        HDF_LOGE("%{public}s:hdi_objcast failed!", __func__);
        return nullptr;
    }

    sptr<ChildType> proxy = OHOS::HDI::hdi_facecast<ChildType>(remote);
    if (proxy == nullptr) {
        HDF_LOGE("%{public}s:hdi_facecast failed!", __func__);
        return nullptr;
    }

    uint32_t serMajorVer = 0;
    uint32_t serMinorVer = 0;
    int32_t pinAuthInterfaceRet = proxy->GetVersion(serMajorVer, serMinorVer);
    if (pinAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s:get version failed!", __func__);
        return nullptr;
    }

    if (serMajorVer != 1) {
        HDF_LOGE("%{public}s:check version failed! version of service:%u.%u, version of client:1.1", __func__, serMajorVer, serMinorVer);
        return nullptr;
    }

    return proxy;
}

sptr<OHOS::HDI::PinAuth::V1_1::IPinAuthInterface> OHOS::HDI::PinAuth::V1_1::IPinAuthInterface::Get(const std::string& serviceName, bool isStub)
{
    if (isStub) {
        std::string desc = Str16ToStr8(OHOS::HDI::PinAuth::V1_1::IPinAuthInterface::GetDescriptor());
        void *impl = LoadHdiImpl(desc.c_str(), serviceName == "pin_auth_interface_service" ? "service" : serviceName.c_str());
        if (impl == nullptr) {
            HDF_LOGE("failed to load hdi impl %{public}s", desc.data());
            return nullptr;
        }
        return reinterpret_cast<OHOS::HDI::PinAuth::V1_1::IPinAuthInterface *>(impl);
    }

    using namespace OHOS::HDI::ServiceManager::V1_0;
    auto servMgr = IServiceManager::Get();
    if (servMgr == nullptr) {
        HDF_LOGE("%{public}s:get IServiceManager failed!", __func__);
        return nullptr;
    }

    sptr<IRemoteObject> remote = servMgr->GetService(serviceName.c_str());
    if (remote == nullptr) {
        HDF_LOGE("%{public}s:get remote object failed!", __func__);
        return nullptr;
    }

    sptr<OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy> proxy = new OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy(remote);
    if (proxy == nullptr) {
        HDF_LOGE("%{public}s:iface_cast failed!", __func__);
        return nullptr;
    }

    proxy->servMgr_ = OHOS::HDI::hdi_objcast<IServiceManager>(servMgr);
    proxy->servMgr_->AddDeathRecipient(
        new OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy::IServiceManagerDeathRecipient(proxy));
    proxy->isReconnected_ = false;
    proxy->serviceName_ = serviceName;
    uint32_t serMajorVer = 0;
    uint32_t serMinorVer = 0;
    int32_t pinAuthInterfaceRet = proxy->GetVersion(serMajorVer, serMinorVer);
    if (pinAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s:get version failed!", __func__);
        return nullptr;
    }

    if (serMajorVer != 1) {
        HDF_LOGE("%{public}s:check version failed! version of service:%u.%u, version of client:1.1", __func__, serMajorVer, serMinorVer);
        return nullptr;
    }

    return proxy;
}

int32_t OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy::Reconnect(
    sptr<OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy> proxy) 
{
    if (proxy == nullptr) {
        HDF_LOGW("Reconnect failed : input proxy is null");
        return HDF_FAILURE;
    }
    using namespace OHOS::HDI::ServiceManager::V1_0;
    proxy->isReconnected_ = false;
    auto iServMgr = IServiceManager::Get();
    if (iServMgr == nullptr) {
        HDF_LOGW("Reconnect failed : iServMgr is null");
        return HDF_FAILURE;
    };
    proxy->reconnectRemote_ = iServMgr->GetService(proxy->serviceName_.c_str());
    if (proxy->reconnectRemote_ == nullptr) {
        HDF_LOGW("Reconnect failed : reconnectRemote_ is null");
        return HDF_FAILURE;
    }
    proxy->servMgr_ = OHOS::HDI::hdi_objcast<IServiceManager>(iServMgr);
    if (proxy->servMgr_ == nullptr) {
        HDF_LOGE("%{public}s:get IServiceManager failed!", __func__);
        return HDF_FAILURE;
    }
    proxy->servMgr_->AddDeathRecipient(
        new OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy::IServiceManagerDeathRecipient(proxy));
    proxy->isReconnected_ = true;
    return HDF_SUCCESS;
}

int32_t OHOS::HDI::PinAuth::V1_1::PinAuthInterfaceProxy::GetExecutorListV1_1_(std::vector<sptr<OHOS::HDI::PinAuth::V1_1::IExecutor>>& executorList, const sptr<IRemoteObject> remote)
{
    MessageParcel pinAuthInterfaceData;
    MessageParcel pinAuthInterfaceReply;
    MessageOption pinAuthInterfaceOption(MessageOption::TF_SYNC);

    if (!pinAuthInterfaceData.WriteInterfaceToken(OHOS::HDI::PinAuth::V1_1::IPinAuthInterface::GetDescriptor())) {
        HDF_LOGE("%{public}s: failed to write interface descriptor!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (!pinAuthInterfaceData.WriteBool(false)) {
        HDF_LOGE("%{public}s:failed to write flag of memory setting!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (remote == nullptr) {
        HDF_LOGE("%{public}s: invalid remote object!", __func__);
        return HDF_ERR_INVALID_OBJECT;
    }

    int32_t pinAuthInterfaceRet = remote->SendRequest(CMD_PIN_AUTH_INTERFACE_GET_EXECUTOR_LIST_V1_1, pinAuthInterfaceData, pinAuthInterfaceReply, pinAuthInterfaceOption);
    if (pinAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, pinAuthInterfaceRet);
        return pinAuthInterfaceRet;
    }

    uint32_t executorListSize = 0;
    if (!pinAuthInterfaceReply.ReadUint32(executorListSize)) {
        HDF_LOGE("%{public}s: failed to read size", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    HDI_CHECK_VALUE_RETURN(executorListSize, >, HDI_BUFF_MAX_SIZE / sizeof(sptr<OHOS::HDI::PinAuth::V1_1::IExecutor>), HDF_ERR_INVALID_PARAM);
    executorList.clear();
    executorList.reserve(executorListSize);
    for (uint32_t i0 = 0; i0 < executorListSize; ++i0) {
        sptr<OHOS::HDI::PinAuth::V1_1::IExecutor> value0;
        sptr<IRemoteObject> value0Remote = pinAuthInterfaceReply.ReadRemoteObject();
        if (value0Remote == nullptr) {
            HDF_LOGE("%{public}s: read an invalid remote object", __func__);
            return HDF_ERR_INVALID_PARAM;
        }

        value0 = new OHOS::HDI::PinAuth::V1_1::ExecutorProxy(value0Remote);
        if (value0 == nullptr) {
            HDF_LOGE("%{public}s: failed to create interface object", __func__);
            return HDF_ERR_INVALID_PARAM;
        }
        executorList.push_back(value0);
    }

    return pinAuthInterfaceRet;
}
