int32_t OHOS::HDI::Power::V1_0::PowerHdiCallbackStub::PowerHdiCallbackStubGetVersion_(MessageParcel& powerHdiCallbackData, MessageParcel& powerHdiCallbackReply, MessageOption& powerHdiCallbackOption, sptr<OHOS::HDI::Power::V1_0::IPowerHdiCallback> impl)
{
    if (powerHdiCallbackData.ReadInterfaceToken() != OHOS::HDI::Power::V1_0::IPowerHdiCallback::GetDescriptor()) {
        HDF_LOGE("%{public}s: interface token check failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    uint32_t majorVer = 0;

    uint32_t minorVer = 0;

    if (impl == nullptr) {
        HDF_LOGE("%{public}s: impl is nullptr!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    int32_t powerHdiCallbackRet = impl->GetVersion(majorVer, minorVer);
    if (powerHdiCallbackRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, powerHdiCallbackRet);
        return powerHdiCallbackRet;
    }

    if (!powerHdiCallbackReply.WriteUint32(majorVer)) {
        HDF_LOGE("%{public}s: write majorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (!powerHdiCallbackReply.WriteUint32(minorVer)) {
        HDF_LOGE("%{public}s: write minorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    return powerHdiCallbackRet;
}
