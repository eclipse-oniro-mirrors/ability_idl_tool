sptr<OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface> OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface::Get(const std::string& serviceName, bool isStub)
{
    if (isStub) {
        std::string desc = Str16ToStr8(OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface::GetDescriptor());
        void *impl = LoadHdiImpl(desc.c_str(), serviceName == "fingerprint_auth_interface_service" ? "service" : serviceName.c_str());
        if (impl == nullptr) {
            HDF_LOGE("failed to load hdi impl %{public}s", desc.data());
            return nullptr;
        }
        return reinterpret_cast<OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface *>(impl);
    }

    using namespace OHOS::HDI::ServiceManager::V1_0;
    auto servMgr = IServiceManager::Get();
    if (servMgr == nullptr) {
        HDF_LOGE("%{public}s:get IServiceManager failed!", __func__);
        return nullptr;
    }

    sptr<IRemoteObject> remote = servMgr->GetService(serviceName.c_str());
    if (remote == nullptr) {
        HDF_LOGE("%{public}s:get remote object failed!", __func__);
        return nullptr;
    }

    sptr<OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy> proxy = new OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy(remote);
    if (proxy == nullptr) {
        HDF_LOGE("%{public}s:iface_cast failed!", __func__);
        return nullptr;
    }

    proxy->servMgr_ = OHOS::HDI::hdi_objcast<IServiceManager>(servMgr);
    proxy->deathRecipient_ = new OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy::IServiceManagerDeathRecipient(proxy);
    proxy->servMgr_->AddDeathRecipient(proxy->deathRecipient_);
    proxy->isReconnected_ = false;
    proxy->serviceName_ = serviceName;
    uint32_t serMajorVer = 0;
    uint32_t serMinorVer = 0;
    int32_t fingerprintAuthInterfaceRet = proxy->GetVersion(serMajorVer, serMinorVer);
    if (fingerprintAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s:get version failed!", __func__);
        return nullptr;
    }

    if (serMajorVer != 1) {
        HDF_LOGE("%{public}s:check version failed! version of service:%u.%u, version of client:1.0", __func__, serMajorVer, serMinorVer);
        return nullptr;
    }

    return proxy;
}

int32_t OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy::Reconnect(
    sptr<OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy> proxy) 
{
    if (proxy == nullptr) {
        HDF_LOGW("Reconnect failed : input proxy is null");
        return HDF_FAILURE;
    }
    using namespace OHOS::HDI::ServiceManager::V1_0;
    proxy->isReconnected_ = false;
    auto iServMgr = IServiceManager::Get();
    if (iServMgr == nullptr) {
        HDF_LOGW("Reconnect failed : iServMgr is null");
        return HDF_FAILURE;
    };
    proxy->reconnectRemote_ = iServMgr->GetService(proxy->serviceName_.c_str());
    if (proxy->reconnectRemote_ == nullptr) {
        HDF_LOGW("Reconnect failed : reconnectRemote_ is null");
        return HDF_FAILURE;
    }
    proxy->servMgr_ = OHOS::HDI::hdi_objcast<IServiceManager>(iServMgr);
    if (proxy->servMgr_ == nullptr) {
        HDF_LOGE("%{public}s:get IServiceManager failed!", __func__);
        return HDF_FAILURE;
    }
    proxy->servMgr_->AddDeathRecipient(
        new OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy::IServiceManagerDeathRecipient(proxy));
    proxy->isReconnected_ = true;
    return HDF_SUCCESS;
}

int32_t OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy::GetExecutorList_(std::vector<sptr<OHOS::HDI::FingerprintAuth::V1_0::IExecutor>>& executorList, const sptr<IRemoteObject> remote)
{
    MessageParcel fingerprintAuthInterfaceData;
    MessageParcel fingerprintAuthInterfaceReply;
    MessageOption fingerprintAuthInterfaceOption(MessageOption::TF_SYNC);

    if (!fingerprintAuthInterfaceData.WriteInterfaceToken(OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface::GetDescriptor())) {
        HDF_LOGE("%{public}s: failed to write interface descriptor!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (!fingerprintAuthInterfaceData.WriteBool(false)) {
        HDF_LOGE("%{public}s:failed to write flag of memory setting!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (remote == nullptr) {
        HDF_LOGE("%{public}s: invalid remote object!", __func__);
        return HDF_ERR_INVALID_OBJECT;
    }

    int32_t fingerprintAuthInterfaceRet = remote->SendRequest(CMD_FINGERPRINT_AUTH_INTERFACE_GET_EXECUTOR_LIST, fingerprintAuthInterfaceData, fingerprintAuthInterfaceReply, fingerprintAuthInterfaceOption);
    if (fingerprintAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, fingerprintAuthInterfaceRet);
        return fingerprintAuthInterfaceRet;
    }

    uint32_t executorListSize = 0;
    if (!fingerprintAuthInterfaceReply.ReadUint32(executorListSize)) {
        HDF_LOGE("%{public}s: failed to read size", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    HDI_CHECK_VALUE_RETURN(executorListSize, >, HDI_BUFF_MAX_SIZE / sizeof(sptr<OHOS::HDI::FingerprintAuth::V1_0::IExecutor>), HDF_ERR_INVALID_PARAM);
    executorList.clear();
    executorList.reserve(executorListSize);
    for (uint32_t i0 = 0; i0 < executorListSize; ++i0) {
        sptr<OHOS::HDI::FingerprintAuth::V1_0::IExecutor> value0;
        sptr<IRemoteObject> value0Remote = fingerprintAuthInterfaceReply.ReadRemoteObject();
        if (value0Remote == nullptr) {
            HDF_LOGE("%{public}s: read an invalid remote object", __func__);
            return HDF_ERR_INVALID_PARAM;
        }

        value0 = new OHOS::HDI::FingerprintAuth::V1_0::ExecutorProxy(value0Remote);
        if (value0 == nullptr) {
            HDF_LOGE("%{public}s: failed to create interface object", __func__);
            return HDF_ERR_INVALID_PARAM;
        }
        executorList.push_back(value0);
    }

    return fingerprintAuthInterfaceRet;
}

int32_t OHOS::HDI::FingerprintAuth::V1_0::FingerprintAuthInterfaceProxy::GetVersion_(uint32_t& majorVer,
     uint32_t& minorVer, const sptr<IRemoteObject> remote)
{
    MessageParcel fingerprintAuthInterfaceData;
    MessageParcel fingerprintAuthInterfaceReply;
    MessageOption fingerprintAuthInterfaceOption(MessageOption::TF_SYNC);

    if (!fingerprintAuthInterfaceData.WriteInterfaceToken(OHOS::HDI::FingerprintAuth::V1_0::IFingerprintAuthInterface::GetDescriptor())) {
        HDF_LOGE("%{public}s: failed to write interface descriptor!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (remote == nullptr) {
        HDF_LOGE("%{public}s: invalid remote object!", __func__);
        return HDF_ERR_INVALID_OBJECT;
    }

    int32_t fingerprintAuthInterfaceRet = remote->SendRequest(CMD_FINGERPRINT_AUTH_INTERFACE_GET_VERSION, fingerprintAuthInterfaceData, fingerprintAuthInterfaceReply, fingerprintAuthInterfaceOption);
    if (fingerprintAuthInterfaceRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, fingerprintAuthInterfaceRet);
        return fingerprintAuthInterfaceRet;
    }

    if (!fingerprintAuthInterfaceReply.ReadUint32(majorVer)) {
        HDF_LOGE("%{public}s: read majorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (!fingerprintAuthInterfaceReply.ReadUint32(minorVer)) {
        HDF_LOGE("%{public}s: read minorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    return fingerprintAuthInterfaceRet;
}
