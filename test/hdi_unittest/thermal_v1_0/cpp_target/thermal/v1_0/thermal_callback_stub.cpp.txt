#include "v1_0/thermal_callback_stub.h"
#include <hdf_base.h>
#include <hdf_core_log.h>

#define HDF_LOG_TAG    thermal_callback_stub

namespace OHOS {
namespace HDI {
namespace Thermal {
namespace V1_0 {



OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStub(const sptr<OHOS::HDI::Thermal::V1_0::IThermalCallback> &impl)
    : IPCObjectStub(OHOS::HDI::Thermal::V1_0::IThermalCallback::GetDescriptor()), impl_(impl)
{
}

OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::~ThermalCallbackStub()
{
    ObjectCollector::GetInstance().RemoveObject(impl_);
}

int32_t OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::OnRemoteRequest(uint32_t code, MessageParcel& data, MessageParcel& reply, MessageOption& option)
{
    switch (code) {
        case CMD_THERMAL_CALLBACK_GET_VERSION:
            return ThermalCallbackStubGetVersion(data, reply, option);
        case CMD_THERMAL_CALLBACK_ON_THERMAL_DATA_EVENT:
            return ThermalCallbackStubOnThermalDataEvent(data, reply, option);
        default: {
            HDF_LOGE("%{public}s: cmd %{public}d is not supported", __func__, code);
            return IPCObjectStub::OnRemoteRequest(code, data, reply, option);
        }
    }
}

int32_t OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubOnThermalDataEvent(MessageParcel& thermalCallbackData, MessageParcel& thermalCallbackReply, MessageOption& thermalCallbackOption)
{
    return OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubOnThermalDataEvent_(thermalCallbackData, thermalCallbackReply, thermalCallbackOption, impl_);
}

int32_t OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubGetVersion(MessageParcel& thermalCallbackData, MessageParcel& thermalCallbackReply, MessageOption& thermalCallbackOption)
{
    return OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubGetVersion_(thermalCallbackData, thermalCallbackReply, thermalCallbackOption, impl_);
}

int32_t OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubOnThermalDataEvent_(MessageParcel& thermalCallbackData, MessageParcel& thermalCallbackReply, MessageOption& thermalCallbackOption, sptr<OHOS::HDI::Thermal::V1_0::IThermalCallback> impl)
{
    if (thermalCallbackData.ReadInterfaceToken() != OHOS::HDI::Thermal::V1_0::IThermalCallback::GetDescriptor()) {
        HDF_LOGE("%{public}s: interface token check failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    OHOS::HDI::Thermal::V1_0::HdfThermalCallbackInfo event;
    if (!HdfThermalCallbackInfoBlockUnmarshalling(thermalCallbackData, event)) {
        HDF_LOGE("%{public}s: read event failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (impl == nullptr) {
        HDF_LOGE("%{public}s: impl is nullptr!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    int32_t thermalCallbackRet = impl->OnThermalDataEvent(event);
    if (thermalCallbackRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, thermalCallbackRet);
        return thermalCallbackRet;
    }

    return thermalCallbackRet;
}

int32_t OHOS::HDI::Thermal::V1_0::ThermalCallbackStub::ThermalCallbackStubGetVersion_(MessageParcel& thermalCallbackData, MessageParcel& thermalCallbackReply, MessageOption& thermalCallbackOption, sptr<OHOS::HDI::Thermal::V1_0::IThermalCallback> impl)
{
    if (thermalCallbackData.ReadInterfaceToken() != OHOS::HDI::Thermal::V1_0::IThermalCallback::GetDescriptor()) {
        HDF_LOGE("%{public}s: interface token check failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    uint32_t majorVer = 0;

    uint32_t minorVer = 0;

    if (impl == nullptr) {
        HDF_LOGE("%{public}s: impl is nullptr!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    int32_t thermalCallbackRet = impl->GetVersion(majorVer, minorVer);
    if (thermalCallbackRet != HDF_SUCCESS) {
        HDF_LOGE("%{public}s failed, error code is %{public}d", __func__, thermalCallbackRet);
        return thermalCallbackRet;
    }

    if (!thermalCallbackReply.WriteUint32(majorVer)) {
        HDF_LOGE("%{public}s: write majorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    if (!thermalCallbackReply.WriteUint32(minorVer)) {
        HDF_LOGE("%{public}s: write minorVer failed!", __func__);
        return HDF_ERR_INVALID_PARAM;
    }

    return thermalCallbackRet;
}
